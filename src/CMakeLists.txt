    kalendarapplication.cpp
    kalendarapplication.h
    main.cpp
    models/monthmodel.cpp

    EXPORT KALENDAR
)

    kalendar_lib
    KF${QT_MAJOR_VERSION}::DBusAddons
    KF${QT_MAJOR_VERSION}::CalendarCore
